Classes

class Programmer {
    constructor (name) {
        this.name = name;
        this.experiences = [];
    }
    code (language) {
        console.log(`Huray! I'm coding with ${language}!`);
        this.experiences.push(language);
    }
    hasExperienceWith (language) {
        return this.experiences.includes(language);
    }
}

const alireza = new Programmer(`Alireza`);
alireza.code(`javascript`);
if (alireza.hasExperienceWith(`javascript`)) console.log(`Hire ${alireza.name} ASAP!`)
